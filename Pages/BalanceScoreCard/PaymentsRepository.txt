public async Task<IEnumerable<Payment>> GetStatementPayments(PaymentRequestModel paymentRequestModel)
{
    var payments = new List<Payment>();
    var response = await httpClient.PostAsJsonAsync($"Payments/GetStatementPayments", paymentRequestModel);
    string contentStr = string.Empty;
    if (response.IsSuccessStatusCode)
    {
        contentStr = await response.Content.ReadAsStringAsync();
        payments = JsonConvert.DeserializeObject<List<Payment>>(contentStr);
    }             
    return payments;            
}
